#include <stdio.h>

int main()
{   int n,x,y=0,z=0,largest,second;
    printf("Enter Number of NO: ");
    scanf("%d",&n);
    for(int  i = 0 ; i < n ; i++) {
        scanf("%d",&x);
        if( y < x ){
            largest = x;
        }
        if( largest > z )
           second = z;
        y = second;
        z = largest;
    }
        printf("%d",second);

    return 0;
}
